 12 import {
 13   GET_PHRASE, PROCESS_GUESS, INPLAY, WON, LOST, RESET_GAME} from './hangman-actions.js';

 15 // get the hangman images array
 16 const images =[];
 17 let e = 0;
 18 [...Array(7)].forEach(() => {

 30 const btnArray = [
 31   'A', 'B', 'C', 'D', 'E',
 32   'F', 'G', 'H', 'I', 'J',
 33   'K', 'L', 'M', 'N', 'O',
 34   'P', 'Q', 'R', 'S', 'T',
 35   'U', 'V', 'W', 'X', 'Y',
 36   'Z', 'RESET'
 37 ]
 38 
 39 const INITIAL_STATE = {
 40   images,
 41   imageIndex: 0,
 42   phrase: {phrase: '', title: ''},
 43   letters: [],
 44   displayPhrase: '',
 45   gameState: INPLAY,
 46   guesses: [],
 47   btnArray,
 48 };
 49 
 50 
 51 const hangman = (state= INITIAL_STATE, action) => {
 52   switch (action.type) {
 53     case GET_PHRASE:
 54       return {
 55         ...state,
 56         phrase: action.phrase,
 57         letters: action.letters,
 58         displayPhrase: action.displayPhrase,
 59       };
 60 
 61     case PROCESS_GUESS:
 62       return {
 63         ...state,
 64         letters: action.letters,
 65         displayPhrase: action.displayPhrase,
 66         guess: action.guess,
 67         imageIndex: action.imageIndex,
 68         gameState: action.gameState,
 69         guesses: action.guesses,
 70       };
 71 
 72     case RESET_GAME:
 73       return {
 74         ...state,
 75         letters: action.letters,
 76         displayPhrase: action.displayPhrase,
 77         guess: action.guess,
 78         imageIndex: action.imageIndex,
 79         gameState: action.gameState,
 80         guesses: action.guesses,
 81         phrase: action.phrase,
 82       };
 83 
 84     default:
 85       return state;
 86   }
 87 };
 88 
 89 export default hangman;
