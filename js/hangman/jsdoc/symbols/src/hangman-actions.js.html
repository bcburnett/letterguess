  3 import Letter from './letter';
  4 import {getRandomPhrase} from './Phrases';
  5 export const GET_PHRASE = 'GET_PHRASE';
  6 export const PROCESS_GUESS = 'PROCESS_GUESS';
  7 export const RESET_GAME = 'RESET_GAME';
  8 export const INPLAY = 'INPLAY';
  9 export const WON = 'CONGRATULATIONS YOU WON!!! Click RESET to play again.';
 10 export const LOST = 'SORRY, YOU LOST. Click RESET to play again.';
 11 const URI = 'http://localhost/~brian/custom-elements/bcb-elements/bcb-hangman/php/phrases.php';
 12 
 13 export const getPhrase = () => async (dispatch) => {
 14   let phrase;
 15   const response = await fetch(URI).catch(()=>phrase={phrase: getRandomPhrase(), title: 'Local Phrases'});
 16   if (response.ok) {
 17     phrase = await response.json();
 18   }
 19   const letters = phrase.phrase.split('').map((e) => new Letter(e));
 20   const displayPhrase = letters.map((e) => e.singleSpacedLetter()).join('');
 21   dispatch({
 22     type: GET_PHRASE,
 23     phrase,
 24     letters,
 25     displayPhrase,
 26   });
 27 };

 29 export const processGuess = (guess, letters, imageIndex, myguesses) => (dispatch) => {
 30   guess = guess.substring(0, 1);
 31   let changed = false;
 32   let gameState = INPLAY;
 33   const guesses = [...myguesses, guess.toUpperCase()];
 34   letters = letters.map((l) => {
 35     if (l.letter.toUpperCase() == guess.toUpperCase()) {
 36       l.correct = true;
 37       changed = true;
 38     }
 39     return l;
 40   });
 41 
 42   if (!changed) imageIndex += 1;
 43   if (imageIndex>6) {
 44     imageIndex = 6;
 45     gameState = LOST;
 46     letters = letters.map((e) => {
 47       e.correct=true;
 48       return e;
 49     });
 50   } else if (letters.every((e)=>e.correct===true)) {
 51     imageIndex = 7;
 52     gameState = WON;
 53   }
 54   const displayPhrase = letters.map((e) => e.singleSpacedLetter()).join('');
 55   dispatch({
 56     type: PROCESS_GUESS,
 57     letters,
 58     displayPhrase,
 59     guess,
 60     imageIndex,
 61     gameState,
 62     guesses,
 63   });
 64 };

 76   const guess = '';
 77   const imageIndex = 0;
 78   const gameState = INPLAY;

 80   dispatch({
